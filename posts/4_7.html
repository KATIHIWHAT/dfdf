def prime(number):
    if number < 2:
        return False
    for i in range(2, int(number)):
        if number % i == 0:
            return False
    return True

def range_sosu(min_num, max_num):
    primes = []
    for number in range(min_num, max_num + 1):
        if prime(number):
            primes.append(number)
    return primes

def rand_sosu(min_num, max_num):
    primes = range_sosu(min_num, max_num)
    return len(primes) / (max_num - min_num + 1)

def prob_sosu(min_num, max_num):
    count = 0
    for number in range(min_num, max_num + 1):
        if range_sosu(number):
            count += 1
    return count / (max_num - min_num + 1)
